function OnEvent(event, arg)
    --OutputLogMessage("Event: "..event.." Arg: "..arg.."\n")
    EnablePrimaryMouseButtonEvents(true);
    if (IsKeyLockOn("numlock") and event == "MOUSE_BUTTON_PRESSED" and arg == 1) then
    ajuste = 0  
      repeat
      --Instrucciones cuando presiono boton izquierdo
      movy = 0
            
      if ajuste<33 then
        movy = math.random(1,2)
      elseif ajuste>33 then
        movy = math.random(2,3)
      end
            
      MoveMouseRelative(0,movy)
      Sleep(18)
      ajuste = ajuste+1
             
      --OutputLogMessage(ajuste.."\n")
      --Fin de instrucciones cuando presionas boton izquierdo
      until not IsMouseButtonPressed(1)
    end
end
